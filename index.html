from itertools import*
def somethinginbetween(F,L):
    limbo = ""
    findex = []
    lindex = []
    screen=["ABC","DEF","GHI"]
    for i in range(3):
        if F in screen[i]:
            findex.append(i)
            findex.append(screen[i].find(F))
    for i in range(3):
        if L in screen[i]:
            lindex.append(i)
            lindex.append(screen[i].find(L))
    if findex[0] == lindex[0]:
        limbo = screen[findex[0]][1]
    elif findex[1] == lindex[1]:
        limbo = screen[1][findex[1]]
    else:
        limbo = "E"
    return limbo

def count_patterns_from(firstPoint, length):
    if length ==0 or length == 1:
        return length
    points = "ABCDEFGHI"
    p_tree = {"A":"BDEFH","B":"ACDEFGI","C":"BDEFH","D":"ABCEGHI","E":"ABCDFGHI","F":"ABCEGHI","G":"BDEFH","H":"ACDEFGI","I":"BDEFH"}
    nice_count = 0
    a_points = points.replace(firstPoint, "")
    for i in permutations(a_points,length-1):
        combo = firstPoint + "".join(i)
        used = [combo[0]]
        bad = False
        for j in range(1, length):
            if combo[j] not in p_tree[combo[j-1]]:
                if somethinginbetween(combo[j], combo[j-1]) not in used:
                    bad = True
                    break
            used.append(combo[j])
        if not bad:
            nice_count += 1
    return nice_count
